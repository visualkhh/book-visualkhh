# 수타입 
```kotlin
fun main() {
  val million = 1_000_000  // Int를 추론
  println(million)
}
/* Output:
1000000
*/

```

```kotlin
fun main() {
  val numerator: Int = 19
  val denominator: Int = 10
  println(numerator % denominator)
}
/* Output:
9
*/

```

```kotlin
fun main() {
  val numerator: Int = 19
  val denominator: Int = 10
  println(numerator / denominator)
}
/* Output:
1
*/

```

```kotlin
fun main() {
  println(45 + 5 * 6)
}
/* Output:
75
*/

```

```kotlin

fun main() {
  println((45 + 5) * 6)
}
/* Output:
300
*/
```

```kotlin
fun main() {
  val i: Int = Int.MAX_VALUE
  println(i + i)
}
/* Output:
-2
*/

```
```kotlin
fun main() {
  val i = 0          // Int 타입을 추론
  val l1 = 0L        // L을 사용해 Long 타입으로
  val l2: Long = 0   // 명시적으로 타입 지정
  println("$l1 $l2")
}
/* Output:
0 0
*/

```

```kotlin
fun main() {
  val i = Int.MAX_VALUE
  println(0L + i + i)              // [1]
  println(1_000_000 * 1_000_000L)  // [2]
}
/* Output:
4294967294
1000000000000
*/

```

```kotlin
fun main() {
  println(Long.MAX_VALUE)
}
/* Output:
9223372036854775807
*/
```
